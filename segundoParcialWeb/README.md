

Base de datos:

Crear primero la base:

```sql
CREATE DATABASE clinica;
```

Entrar a la base `clinica` y ejecutar:

```sql
CREATE EXTENSION IF NOT EXISTS btree_gist;

CREATE TABLE IF NOT EXISTS roles (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    nombre VARCHAR(30) NOT NULL UNIQUE
);

CREATE TABLE IF NOT EXISTS usuarios (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    username VARCHAR(40) NOT NULL UNIQUE,
    password VARCHAR(255) NOT NULL,
    enabled BOOLEAN NOT NULL
);

CREATE TABLE IF NOT EXISTS usuario_roles (
    usuario_id BIGINT NOT NULL,
    rol_id BIGINT NOT NULL,
    PRIMARY KEY (usuario_id, rol_id),
    CONSTRAINT fk_usuario_roles_usuario FOREIGN KEY (usuario_id) REFERENCES usuarios(id),
    CONSTRAINT fk_usuario_roles_rol FOREIGN KEY (rol_id) REFERENCES roles(id)
);

CREATE TABLE IF NOT EXISTS medicos (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    nombre VARCHAR(40) NOT NULL,
    especialidad VARCHAR(60) NOT NULL,
    documento VARCHAR(20) NOT NULL UNIQUE,
    usuario_id BIGINT UNIQUE,
    CONSTRAINT fk_medicos_usuario FOREIGN KEY (usuario_id) REFERENCES usuarios(id)
);

CREATE TABLE IF NOT EXISTS pacientes (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    nombre VARCHAR(40) NOT NULL,
    documento VARCHAR(20) NOT NULL UNIQUE,
    usuario_id BIGINT UNIQUE,
    CONSTRAINT fk_pacientes_usuario FOREIGN KEY (usuario_id) REFERENCES usuarios(id)
);

CREATE TABLE IF NOT EXISTS consultorios (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    numero INTEGER NOT NULL UNIQUE
);

CREATE TABLE IF NOT EXISTS consultas (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    motivo VARCHAR(100) NOT NULL,
    fecha DATE NOT NULL,
    hora_inicio TIME NOT NULL,
    hora_fin TIME NOT NULL,
    paciente_id BIGINT NOT NULL,
    medico_id BIGINT NOT NULL,
    consultorio_id BIGINT NOT NULL,
    CONSTRAINT chk_consulta_horario CHECK (hora_inicio < hora_fin),
    CONSTRAINT fk_consultas_paciente FOREIGN KEY (paciente_id) REFERENCES pacientes(id),
    CONSTRAINT fk_consultas_medico FOREIGN KEY (medico_id) REFERENCES medicos(id),
    CONSTRAINT fk_consultas_consultorio FOREIGN KEY (consultorio_id) REFERENCES consultorios(id)
);

DO $$
BEGIN

        SELECT 1 FROM pg_constraint WHERE conname = 'consultas_consultorio_sin_cruce'
    ) THEN
        ALTER TABLE consultas

            EXCLUDE USING gist (
                consultorio_id WITH =,
                tsrange(fecha + hora_inicio, fecha + hora_fin, '[)') WITH &&
            );
    END IF;
END $$;

DO $$
BEGIN
    IF NOT EXISTS (
        SELECT 1 FROM pg_constraint WHERE conname = 'consultas_medico_sin_cruce'
    ) THEN
        ALTER TABLE consultas
            ADD CONSTRAINT consultas_medico_sin_cruce
            EXCLUDE USING gist (
                medico_id WITH =,
                tsrange(fecha + hora_inicio, fecha + hora_fin, '[)') WITH &&
            );
    END IF;
END $$;
```

## Configuracion

La conexion por defecto esta en `src/main/resources/application.properties`:

```properties
spring.datasource.url=jdbc:postgresql://localhost:5432/clinica
spring.datasource.username=postgres
spring.datasource.password=1234
spring.sql.init.mode=never
```

Si la clave de PostgreSQL es diferente, cambiar `spring.datasource.password`.

## Ejecutar

```bash
mvnw.cmd spring-boot:run
```

Aplicacion:

```text
http://localhost:8080
```

Swagger:

```text
http://localhost:8080/swagger-ui.html
```

Si el puerto `8080` esta ocupado:

```bash
netstat -ano | findstr :8080
taskkill /PID NUMERO_DEL_PID /F
```
